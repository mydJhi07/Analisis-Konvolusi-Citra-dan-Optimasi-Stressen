#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int **alloc_m(int n) {
    int **m = (int **)malloc(n * sizeof(int *));
    for (int i = 0; i < n; i++) m[i] = (int *)calloc(n, sizeof(int));
    return m;
}

void free_m(int **m, int n) {
    for (int i = 0; i < n; i++) free(m[i]);
    free(m);
}

void init_rand(int **m, int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) m[i][j] = rand() % 10;
}

void add_m(int **A, int **B, int **C, int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) C[i][j] = A[i][j] + B[i][j];
}

void sub_m(int **A, int **B, int **C, int n) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) C[i][j] = A[i][j] - B[i][j];
}

void standard_mult(int **A, int **B, int **C, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            C[i][j] = 0;
            for (int k = 0; k < n; k++) C[i][j] += A[i][k] * B[k][j];
        }
    }
}

void strassen(int **A, int **B, int **C, int n) {
    if (n <= 64) {
        standard_mult(A, B, C, n);
        return;
    }

    int new_n = n / 2;
    int **A00 = alloc_m(new_n); int **A01 = alloc_m(new_n);
    int **A10 = alloc_m(new_n); int **A11 = alloc_m(new_n);
    int **B00 = alloc_m(new_n); int **B01 = alloc_m(new_n);
    int **B10 = alloc_m(new_n); int **B11 = alloc_m(new_n);
    int **C00 = alloc_m(new_n); int **C01 = alloc_m(new_n);
    int **C10 = alloc_m(new_n); int **C11 = alloc_m(new_n);
    int **M1 = alloc_m(new_n); int **M2 = alloc_m(new_n);
    int **M3 = alloc_m(new_n); int **M4 = alloc_m(new_n);
    int **M5 = alloc_m(new_n); int **M6 = alloc_m(new_n);
    int **M7 = alloc_m(new_n);
    int **tempA = alloc_m(new_n); int **tempB = alloc_m(new_n);

    for (int i = 0; i < new_n; i++) {
        for (int j = 0; j < new_n; j++) {
            A00[i][j] = A[i][j]; A01[i][j] = A[i][j + new_n];
            A10[i][j] = A[i + new_n][j]; A11[i][j] = A[i + new_n][j + new_n];
            B00[i][j] = B[i][j]; B01[i][j] = B[i][j + new_n];
            B10[i][j] = B[i + new_n][j]; B11[i][j] = B[i + new_n][j + new_n];
        }
    }

    add_m(A00, A11, tempA, new_n); add_m(B00, B11, tempB, new_n); strassen(tempA, tempB, M1, new_n);
    add_m(A10, A11, tempA, new_n); strassen(tempA, B00, M2, new_n);
    sub_m(B01, B11, tempB, new_n); strassen(A00, tempB, M3, new_n);
    sub_m(B10, B00, tempB, new_n); strassen(A11, tempB, M4, new_n);
    add_m(A00, A01, tempA, new_n); strassen(tempA, B11, M5, new_n);
    sub_m(A10, A00, tempA, new_n); add_m(B00, B01, tempB, new_n); strassen(tempA, tempB, M6, new_n);
    sub_m(A01, A11, tempA, new_n); add_m(B10, B11, tempB, new_n); strassen(tempA, tempB, M7, new_n);

    add_m(M1, M4, tempA, new_n); sub_m(tempA, M5, tempB, new_n); add_m(tempB, M7, C00, new_n);
    add_m(M3, M5, C01, new_n);
    add_m(M2, M4, C10, new_n);
    sub_m(M1, M2, tempA, new_n); add_m(tempA, M3, tempB, new_n); add_m(tempB, M6, C11, new_n);

    for (int i = 0; i < new_n; i++) {
        for (int j = 0; j < new_n; j++) {
            C[i][j] = C00[i][j]; C[i][j + new_n] = C01[i][j];
            C[i + new_n][j] = C10[i][j]; C[i + new_n][j + new_n] = C11[i][j];
        }
    }

    free_m(A00, new_n); free_m(A01, new_n); free_m(A10, new_n); free_m(A11, new_n);
    free_m(B00, new_n); free_m(B01, new_n); free_m(B10, new_n); free_m(B11, new_n);
    free_m(C00, new_n); free_m(C01, new_n); free_m(C10, new_n); free_m(C11, new_n);
    free_m(M1, new_n); free_m(M2, new_n); free_m(M3, new_n); free_m(M4, new_n);
    free_m(M5, new_n); free_m(M6, new_n); free_m(M7, new_n);
    free_m(tempA, new_n); free_m(tempB, new_n);
}

void run_performance_test(int n) {
    int **A = alloc_m(n); int **B = alloc_m(n); int **C = alloc_m(n);
    init_rand(A, n); init_rand(B, n);

    clock_t start = clock();
    strassen(A, B, C, n);
    clock_t end = clock();

    double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("Waktu eksekusi Strassen N=%d : %f detik\n", n, time_taken);

    free_m(A, n); free_m(B, n); free_m(C, n);
}

int main() {
    srand(time(NULL));
    printf("--- Pengujian Performa Strassen ---\n");
    run_performance_test(512);
    run_performance_test(1024);
    return 0;
}
