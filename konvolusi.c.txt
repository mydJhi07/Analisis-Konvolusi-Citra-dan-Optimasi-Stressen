#include <stdio.h>
#include <stdlib.h>

#define STB_IMAGE_IMPLEMENTATION
#include "stb_image.h"
#define STB_IMAGE_WRITE_IMPLEMENTATION
#include "stb_image_write.h"

void apply_blur(unsigned char *in, unsigned char *out, int w, int h, int c, int k_size) {
    int r = k_size / 2;
    for (int y = 0; y < h; y++) {
        for (int x = 0; x < w; x++) {
            for (int ch = 0; ch < c; ch++) {
                float sum = 0; int count = 0;
                for (int ky = -r; ky <= r; ky++) {
                    for (int kx = -r; kx <= r; kx++) {
                        int py = y + ky; int px = x + kx;
                        if (px >= 0 && px < w && py >= 0 && py < h) {
                            sum += in[(py * w + px) * c + ch];
                            count++;
                        }
                    }
                }
                out[(y * w + x) * c + ch] = (unsigned char)(sum / count);
            }
        }
    }
}

int main() {
    int width, height, channels;
    
    // Ganti "input.jpg" dengan nama gambar yang ada di komputer Anda
    printf("Memuat gambar...\n");
    unsigned char *img_in = stbi_load("input.jpg", &width, &height, &channels, 0);
    
    if(img_in == NULL) {
        printf("Gagal memuat gambar. Pastikan file gambar tersedia.\n");
        return 1;
    }

    unsigned char *img_out = malloc(width * height * channels);
    int kernel_size = 5; // Ukuran kernel (misal 5x5)

    printf("Memproses blurring pada gambar berukuran %dx%d (%d channels)...\n", width, height, channels);
    apply_blur(img_in, img_out, width, height, channels, kernel_size);

    stbi_write_png("output_blur.png", width, height, channels, img_out, width * channels);
    printf("Proses selesai! Hasil disimpan sebagai output_blur.png\n");

    stbi_image_free(img_in);
    free(img_out);
    return 0;
}
