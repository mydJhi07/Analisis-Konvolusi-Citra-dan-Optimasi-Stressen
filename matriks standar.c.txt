#include <stdio.h>
#include <stdlib.h>

int main() {
    int n = 3; // Contoh ukuran matriks

    // Mengalokasikan memori secara dinamis untuk A, B, dan C (Heap)
    int **A = (int **)malloc(n * sizeof(int *));
    int **B = (int **)malloc(n * sizeof(int *));
    int **C = (int **)malloc(n * sizeof(int *));
    for(int i = 0; i < n; i++) {
        A[i] = (int *)malloc(n * sizeof(int));
        B[i] = (int *)malloc(n * sizeof(int));
        C[i] = (int *)malloc(n * sizeof(int));
    }

    // Inisialisasi data matriks A dan B
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            A[i][j] = i + j;
            B[i][j] = i * j;
        }
    }

    // Algoritma Perkalian Matriks O(n^3) - Tiga Loop Bersarang
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            C[i][j] = 0;
            for (int k = 0; k < n; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }

    // Cetak Hasil
    printf("Hasil Perkalian Matriks Modul 1:\n");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            printf("%d ", C[i][j]);
        }
        printf("\n");
    }

    // Wajib: Bebaskan memori untuk mencegah Memory Leak
    for(int i = 0; i < n; i++) {
        free(A[i]); free(B[i]); free(C[i]);
    }
    free(A); free(B); free(C);

    return 0;
}
